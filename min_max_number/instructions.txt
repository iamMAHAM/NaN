FONCTION MIN(TABLEAU)
    SI TYPE(TABLEAU) EST LIST AND TAILLE TABLEAU DIFFERENT 0
        POUR ELEMENT TABLEAU
            SI ELEMENT DIFFERENT REEL
                RETOURNER -1
        
        MIN = TABLEAU[0]

        POUR ELEMENT TABLEAU
            SI ELEMENT INFERIEUR MIN
                MIN = ELEMENT
        RETOURNE MIN
        
    RETOURNE -1


FONCTION MAX(TABLEAU)
    SI TYPE(TABLEAU) EST LIST AND TAILLE TABLEAU DIFFERENT 0
        POUR ELEMENT TABLEAU
            SI ELEMENT DIFFERENT REEL
                RETOURNER -1
        
        MIN = TABLEAU[0]

        POUR ELEMENT TABLEAU
            SI ELEMENT SUPERIEUR MIN
                MIN = ELEMENT
        RETOURNE MIN
        
    RETOURNE -1


FONCTION MIN_MAX(TABLEAU)
    RETOURNE (MIN(TABLEAU), MAX(TABLEAU))
